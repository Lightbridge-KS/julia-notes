6.1 IF

if *condition 1*
    *option 1*
elseif *condition 2*
    *option 2*
else
    *option 3*
end
function FizzBuzz(🥚)
    
    if (🥚 % 3 == 0) && (🥚 % 5 == 0)
        println("Fizz Buzz 🚌")
    elseif 🥚 % 3 == 0
        println("$(🥚 ÷ 3) Fizz")
    elseif 🥚 % 5 == 0
        println("$(🥚 ÷ 5) Buzz 🚌")
    else
        println("$🥚")
    end

end
FizzBuzz (generic function with 1 method)
FizzBuzz(15)
Fizz Buzz 🚌

Boolean in if must be true or false.

if 1
    println("true")
end
# This will error

6.2 Ternary operators

a ? b : c

Translate to

if a
    b
else
    c
end
what_is_larger(x, y) = x > y ? "$x is larger than $y" : "$y is larger than $x" 

what_is_larger(1, 2)
"2 is larger than 1"
